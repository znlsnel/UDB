🌊 Gerstner 파도 구현

기본 파도 셰이더

Shader "Water/GerstnerWaves" { Properties { _WaterColor ("Water Color", Color) = (0.1, 0.4, 0.7, 0.8) _DeepWaterColor ("Deep Water Color", Color) = (0.0, 0.2, 0.5, 1.0) _WaveAmplitude ("Wave Amplitude", Range(0, 2)) = 0.5 _WaveFrequency ("Wave Frequency", Range(0, 10)) = 2.0 _WaveSpeed ("Wave Speed", Range(0, 5)) = 1.0 _WaveDirection ("Wave Direction", Vector) = (1, 0, 0, 0) _NormalMap ("Normal Map", 2D) = "bump" {} _Reflection ("Reflection", Cube) = "_Skybox" {} } SubShader { Tags { "RenderType"="Transparent" "Queue"="Transparent" } LOD 200 Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" struct appdata { float4 vertex : POSITION; float3 normal : NORMAL; float2 uv : TEXCOORD0; }; struct v2f { float4 pos : SV_POSITION; float3 worldPos : TEXCOORD0; float3 worldNormal : TEXCOORD1; float2 uv : TEXCOORD2; float4 screenPos : TEXCOORD3; }; float4 _WaterColor, _DeepWaterColor; float _WaveAmplitude, _WaveFrequency, _WaveSpeed; float4 _WaveDirection; sampler2D _NormalMap; samplerCUBE _Reflection; sampler2D _CameraDepthTexture; // Gerstner 파도 함수 float3 GerstnerWave(float2 pos, float amplitude, float frequency, float speed, float2 direction) { float phase = frequency * dot(direction, pos) + speed * _Time.y; float steepness = 0.5; // 파도의 가파름 float3 wave; wave.x = steepness * amplitude * direction.x * cos(phase); wave.z = steepness * amplitude * direction.y * cos(phase); wave.y = amplitude * sin(phase); return wave; } // 다중 파도 합성 float3 CalculateWaves(float2 worldPos) { float3 waves = 0; // 주 파도 waves += GerstnerWave(worldPos, _WaveAmplitude, _WaveFrequency, _WaveSpeed, _WaveDirection.xy); // 보조 파도들 waves += GerstnerWave(worldPos, _WaveAmplitude * 0.5, _WaveFrequency * 1.5, _WaveSpeed * 1.2, float2(0.7, 0.7)); waves += GerstnerWave(worldPos, _WaveAmplitude * 0.3, _WaveFrequency * 2.0, _WaveSpeed * 0.8, float2(-0.5, 0.8)); return waves; } v2f vert(appdata v) { v2f o; // 월드 좌표 계산 float3 worldPos = mul(unity_ObjectToWorld, v.vertex).xyz; // 파도 변위 적용 float3 waveOffset = CalculateWaves(worldPos.xz); worldPos += waveOffset; o.pos = mul(UNITY_MATRIX_VP, float4(worldPos, 1.0)); o.worldPos = worldPos; o.worldNormal = UnityObjectToWorldNormal(v.normal); o.uv = v.uv; o.screenPos = ComputeScreenPos(o.pos); return o; } float4 frag(v2f i) : SV_Target { // 깊이 기반 색상 float depth = LinearEyeDepth(tex2Dproj(_CameraDepthTexture, i.screenPos).r); float waterDepth = depth - i.screenPos.w; float depthFactor = saturate(waterDepth / 5.0); // 물 색상 보간 float4 waterColor = lerp(_WaterColor, _DeepWaterColor, depthFactor); // 노멀 맵 샘플링 float2 normalUV = i.uv + _Time.y * 0.1; float3 normal = UnpackNormal(tex2D(_NormalMap, normalUV)); // 반사 계산 float3 viewDir = normalize(_WorldSpaceCameraPos - i.worldPos); float3 reflectDir = reflect(-viewDir, normal); float4 reflection = texCUBE(_Reflection, reflectDir); // 프레넬 효과 float fresnel = pow(1.0 - saturate(dot(normal, viewDir)), 2.0); // 최종 색상 합성 float4 finalColor = lerp(waterColor, reflection, fresnel * 0.8); finalColor.a = waterColor.a; return finalColor; } ENDHLSL } } }

🫧 거품과 폼 효과

물의 가장자리와 파도 정점에서 생성되는 자연스러운 거품 효과를 구현해보겠습니다.

깊이 기반 폼 셰이더

// 거품 효과를 위한 추가 프래그먼트 셰이더 코드 float4 frag(v2f i) : SV_Target { // 기본 물 색상 계산 (이전 코드와 동일) float depth = LinearEyeDepth(tex2Dproj(_CameraDepthTexture, i.screenPos).r); float waterDepth = depth - i.screenPos.w; // 거품 효과 계산 float foamDepth = saturate(waterDepth / 0.5); // 얕은 곳에서 거품 float foamMask = 1.0 - foamDepth; // 노이즈 텍스처로 거품 패턴 생성 float2 foamUV = i.uv * 10.0 + _Time.y * float2(0.1, 0.05); float foamNoise = tex2D(_FoamTexture, foamUV).r; // 파도 정점에서의 거품 float waveHeight = i.worldPos.y - _WaterLevel; float waveFoam = saturate(waveHeight / _WaveAmplitude - 0.7); // 거품 강도 계산 float foamIntensity = saturate(foamMask + waveFoam) * foamNoise; // 거품 색상 적용 float3 foamColor = float3(1.0, 1.0, 1.0); float4 waterColor = CalculateWaterColor(i); // 기본 물 색상 // 최종 색상 블렌딩 float4 finalColor = lerp(waterColor, float4(foamColor, 1.0), foamIntensity); return finalColor; }

코스틱 효과 구현

// 코스틱 (물 바닥 빛 무늬) 효과 float CalculateCaustics(float2 worldPos, float depth) { // 다중 스케일 노이즈로 코스틱 패턴 생성 float2 causticsUV1 = worldPos * 0.5 + _Time.y * float2(0.03, 0.02); float2 causticsUV2 = worldPos * 0.8 + _Time.y * float2(-0.02, 0.04); float caustics1 = tex2D(_CausticsTexture, causticsUV1).r; float caustics2 = tex2D(_CausticsTexture, causticsUV2).r; // 두 패턴을 곱해서 더 복잡한 무늬 생성 float caustics = caustics1 * caustics2; // 깊이에 따른 강도 조절 float depthFade = exp(-depth * 0.5); caustics *= depthFade; // 대비 증가 caustics = pow(caustics, 2.0) * 2.0; return saturate(caustics); } // 수중 오브젝트에 코스틱 적용 float4 ApplyCausticsToUnderwater(float4 baseColor, float3 worldPos) { float caustics = CalculateCaustics(worldPos.xz, worldPos.y); // 코스틱을 라이팅에 추가 float3 causticsColor = float3(0.8, 1.0, 1.2) * caustics; baseColor.rgb += causticsColor * _CausticsIntensity; return baseColor; }

🎮 상호작용 시스템

플레이어나 오브젝트와 물의 상호작용을 구현하여 더욱 몰입감 있는 경험을 제공합니다.

물체 충돌 감지 및 파문 생성

// C# 스크립트: 물 상호작용 매니저 using UnityEngine; using System.Collections.Generic; public class WaterInteractionManager : MonoBehaviour { public Material waterMaterial; public LayerMask interactionLayers = -1; public float rippleStrength = 1.0f; public float rippleSpeed = 2.0f; private List<RippleData> activeRipples = new List<RippleData>(); private Vector4[] ripplePositions = new Vector4[10]; private Vector4[] rippleData = new Vector4[10]; struct RippleData { public Vector3 position; public float startTime; public float strength; public float maxRadius; } void Update() { UpdateRipples(); SendRipplesToShader(); } public void CreateRipple(Vector3 position, float strength = 1.0f) { if (activeRipples.Count >= 10) return; RippleData newRipple = new RippleData { position = position, startTime = Time.time, strength = strength * rippleStrength, maxRadius = 5.0f }; activeRipples.Add(newRipple); } void UpdateRipples() { for (int i = activeRipples.Count - 1; i >= 0; i--) { RippleData ripple = activeRipples[i]; float age = Time.time - ripple.startTime; float radius = age * rippleSpeed; if (radius > ripple.maxRadius) { activeRipples.RemoveAt(i); continue; } // 셰이더로 전달할 데이터 준비 ripplePositions[i] = new Vector4(ripple.position.x, ripple.position.z, radius, 0); rippleData[i] = new Vector4(ripple.strength, ripple.maxRadius, age, 0); } } void SendRipplesToShader() { waterMaterial.SetVectorArray("_RipplePositions", ripplePositions); waterMaterial.SetVectorArray("_RippleData", rippleData); waterMaterial.SetInt("_RippleCount", activeRipples.Count); } void OnTriggerEnter(Collider other) { if (((1 << other.gameObject.layer) & interactionLayers) != 0) { Vector3 impactPoint = other.ClosestPoint(transform.position); float impactStrength = other.attachedRigidbody ? other.attachedRigidbody.velocity.magnitude : 1.0f; CreateRipple(impactPoint, impactStrength); } } }

셰이더에서 파문 처리

// 셰이더에서 동적 파문 처리 float4 _RipplePositions[10]; float4 _RippleData[10]; int _RippleCount; float CalculateRippleEffect(float2 worldPos) { float totalRipple = 0.0; for (int i = 0; i < _RippleCount; i++) { float2 ripplePos = _RipplePositions[i].xy; float currentRadius = _RipplePositions[i].z; float strength = _RippleData[i].x; float maxRadius = _RippleData[i].y; float age = _RippleData[i].z; float distance = length(worldPos - ripplePos); // 링 형태의 파문 float ringWidth = 0.5; float ringFactor = exp(-pow((distance - currentRadius) / ringWidth, 2.0)); // 시간에 따른 감쇠 float decay = exp(-age * 2.0); totalRipple += ringFactor * strength * decay; } return totalRipple; } // 버텍스 셰이더에서 파문 적용 v2f vert(appdata v) { v2f o; float3 worldPos = mul(unity_ObjectToWorld, v.vertex).xyz; // 기본 파도 + 동적 파문 float3 waveOffset = CalculateWaves(worldPos.xz); float rippleOffset = CalculateRippleEffect(worldPos.xz); worldPos.y += waveOffset.y + rippleOffset; worldPos.xz += waveOffset.xz; o.pos = mul(UNITY_MATRIX_VP, float4(worldPos, 1.0)); // ... 나머지 출력 설정 return o; }

💡 물 셰이더 최적화 팁:
  • LOD 시스템: 거리별로 파도 복잡도 조절
  • 텍스처 아틀라스: 노멀맵, 폼, 코스틱을 하나로 합치기
  • 인스턴싱: 여러 물 영역을 효율적으로 렌더링
  • 컬링: 화면 밖 물 영역 제거
  • 모바일 최적화: 복잡한 계산을 버텍스 셰이더로 이동
⚠️ 주의사항:
  • 성능 모니터링: 복잡한 물 셰이더는 GPU 부하가 클 수 있음
  • 플랫폼별 조정: 모바일과 PC에서 다른 품질 설정 필요
  • 물리 시뮬레이션: 부력 계산 시 물 높이 정확히 계산
  • 라이팅 일관성: 물 표면과 수중 라이팅 통일
