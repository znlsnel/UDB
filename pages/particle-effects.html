                    #endif
                    
                    o.color = v.color * _TintColor;
                    o.texcoord = TRANSFORM_TEX(v.texcoord, _MainTex);

                    return o;
                }

                    float fade = saturate(_InvFade * (sceneZ - partZ));
                    i.color.a *= fade;
                    #endif
                    
                    fixed4 col = 2.0f * i.color * tex2D(_MainTex, i.texcoord);
                    UNITY_APPLY_FOG(i.fogCoord, col);
                    return col;
                }
                ENDCG
            }
        }
    }
}

Shader "Custom/MagicParticle" {
    Properties {
        _MainTex ("Particle Texture", 2D) = "white" {}
        _GlowColor ("Glow Color", Color) = (0, 0.5, 1, 1)
        _PulseSpeed ("Pulse Speed", Range(0.1, 5)) = 2
        _GlowIntensity ("Glow Intensity", Range(1, 10)) = 3
        _Fresnel ("Fresnel Power", Range(0.1, 5)) = 2
    }
    
    SubShader {
        Tags { 
            "Queue"="Transparent" 
            "RenderType"="Transparent" 
        }
        
        Blend One One  // Additive blending
        ZWrite Off
        Cull Off
        
        Pass {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            
            struct appdata {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
                float3 normal : NORMAL;
                fixed4 color : COLOR;
            };
            
            struct v2f {
                float2 uv : TEXCOORD0;
                float4 vertex : SV_POSITION;
                float3 worldNormal : TEXCOORD1;
                float3 viewDir : TEXCOORD2;
                fixed4 color : COLOR;
            };
            
            sampler2D _MainTex;
            fixed4 _GlowColor;
            float _PulseSpeed;
            float _GlowIntensity;
            float _Fresnel;
            
            v2f vert (appdata v) {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = v.uv;
                o.worldNormal = UnityObjectToWorldNormal(v.normal);
                o.viewDir = normalize(WorldSpaceViewDir(v.vertex));
                o.color = v.color;
                return o;
            }
            
            fixed4 frag (v2f i) : SV_Target {
                // ë©”ì¸ í…ìŠ¤ì²˜
                fixed4 tex = tex2D(_MainTex, i.uv);
                
                // í„ìŠ¤ íš¨ê³¼
                float pulse = sin(_Time.y * _PulseSpeed) * 0.5 + 0.5;
                
                // í”„ë ˆë„¬ íš¨ê³¼
                float fresnel = pow(1.0 - dot(i.worldNormal, i.viewDir), _Fresnel);
                
                // ìµœì¢… ìƒ‰ìƒ ê³„ì‚°
                fixed4 col = tex * _GlowColor * i.color;
                col.rgb *= _GlowIntensity * (1 + pulse) * (1 + fresnel);
                
                return col;
            }
            ENDCG
        }
    }
}
