🎨 툰 렌더링의 매력

셀 셰이딩, 아웃라인, 스타일라이즈드 라이팅으로 만드는 애니메이션 같은 비주얼

🎭 툰 셰이더의 핵심 요소

툰 셰이더는 3D 모델을 2D 애니메이션처럼 보이게 만드는 비사실적 렌더링(NPR) 기법입니다. 일본 애니메이션, 만화, 게임에서 널리 사용되는 스타일입니다.

🌅 사실적 셰이딩

부드러운 그라디언트와 자연스러운 명암

🎨 툰 셰이딩

명확한 색상 구분과 단계적 명암

Cell

셀 셰이딩

단계적 명암: 부드러운 그라디언트 대신 명확한 색상 구분

램프 텍스처: 커스텀 라이팅 곡선

색상 밴딩: 제한된 색상 팔레트

하드 라이팅: 날카로운 그림자 경계

Outline

아웃라인

외곽선 렌더링: 오브젝트 경계선 강조

다중 패스: 백페이스 확장 기법

두께 조절: 거리별 선 굵기 변화

색상 커스터마이징: 상황별 아웃라인 색상

Rim

림 라이팅

가장자리 조명: 오브젝트 윤곽 강조

프레넬 효과: 시야각 기반 강도

색상 그라디언트: 다채로운 림 효과

애니메이션 효과: 펄스, 글로우 등

Gradient

그라디언트 맵

색상 리맵핑: 명도를 색상으로 변환

아티스틱 컨트롤: 완전한 색상 제어

분위기 연출: 시간대별 색감 변화

스타일 통일: 일관된 비주얼 톤

Stylized

스타일라이즈드 효과

하프톤 패턴: 만화책 스타일 음영

크로스 해칭: 펜화 스타일 표현

포스터라이제이션: 색상 단계 제한

🎨 기본 셀 셰이딩 구현

가장 기본적인 툰 셰이더인 셀 셰이딩을 구현해보겠습니다.

기본 셀 셰이더

Shader "Toon/BasicCellShading" { Properties { _MainTex ("Main Texture", 2D) = "white" {} _Color ("Color", Color) = (1, 1, 1, 1) _ShadowColor ("Shadow Color", Color) = (0.5, 0.5, 0.5, 1) _ShadowThreshold ("Shadow Threshold", Range(0, 1)) = 0.5 _ShadowSmoothness ("Shadow Smoothness", Range(0, 1)) = 0.1 _RampTex ("Ramp Texture", 2D) = "white" {} } SubShader { Tags { "RenderType"="Opaque" "LightMode"="ForwardBase" } LOD 200 Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" #include "Lighting.cginc" struct appdata { float4 vertex : POSITION; float3 normal : NORMAL; float2 uv : TEXCOORD0; }; struct v2f { float4 pos : SV_POSITION; float2 uv : TEXCOORD0; float3 worldNormal : TEXCOORD1; float3 worldPos : TEXCOORD2; }; sampler2D _MainTex; sampler2D _RampTex; float4 _MainTex_ST; float4 _Color; float4 _ShadowColor; float _ShadowThreshold; float _ShadowSmoothness; v2f vert(appdata v) { v2f o; o.pos = UnityObjectToClipPos(v.vertex); o.uv = TRANSFORM_TEX(v.uv, _MainTex); o.worldNormal = UnityObjectToWorldNormal(v.normal); o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz; return o; } float4 frag(v2f i) : SV_Target { // 기본 텍스처 샘플링 float4 albedo = tex2D(_MainTex, i.uv) * _Color; // 라이팅 계산 float3 normal = normalize(i.worldNormal); float3 lightDir = normalize(_WorldSpaceLightPos0.xyz); float NdotL = dot(normal, lightDir); // 셀 셰이딩을 위한 단계화 float lightIntensity = smoothstep(_ShadowThreshold - _ShadowSmoothness, _ShadowThreshold + _ShadowSmoothness, NdotL * 0.5 + 0.5); // 램프 텍스처 사용 (선택적) float rampValue = tex2D(_RampTex, float2(lightIntensity, 0.5)).r; // 최종 색상 계산 float3 lightColor = lerp(_ShadowColor.rgb, float3(1, 1, 1), rampValue); float3 finalColor = albedo.rgb * lightColor * _LightColor0.rgb; // 앰비언트 라이팅 추가 finalColor += albedo.rgb * UNITY_LIGHTMODEL_AMBIENT.rgb; return float4(finalColor, albedo.a); } ENDHLSL } } }

✏️ 아웃라인 렌더링

1

2

포스트 프로세싱

깊이와 노멀 버퍼를 이용한 에지 디텍션

3

실시간으로 아웃라인 지오메트리 생성

4

스텐실 버퍼

스텐실을 이용한 정밀한 아웃라인 제어

백페이스 확장 아웃라인

Shader "Toon/OutlineShader" { Properties { _MainTex ("Main Texture", 2D) = "white" {} _Color ("Color", Color) = (1, 1, 1, 1) _OutlineColor ("Outline Color", Color) = (0, 0, 0, 1) _OutlineWidth ("Outline Width", Range(0, 0.1)) = 0.01 _OutlineDistanceFade ("Distance Fade", Range(0, 1)) = 1 } SubShader { Tags { "RenderType"="Opaque" } // 아웃라인 패스 (첫 번째) Pass { Name "OUTLINE" Cull Front ZWrite On ColorMask RGB HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" struct appdata { float4 vertex : POSITION; float3 normal : NORMAL; }; struct v2f { float4 pos : SV_POSITION; }; float _OutlineWidth; float4 _OutlineColor; float _OutlineDistanceFade; v2f vert(appdata v) { v2f o; // 노멀 방향으로 버텍스 확장 float3 norm = normalize(v.normal); float3 worldPos = mul(unity_ObjectToWorld, v.vertex).xyz; // 거리 기반 아웃라인 두께 조절 float distance = length(_WorldSpaceCameraPos - worldPos); float distanceFade = lerp(1, 1 / distance, _OutlineDistanceFade); // 아웃라인 확장 float4 clipPosition = UnityObjectToClipPos(v.vertex); float3 clipNormal = mul((float3x3) UNITY_MATRIX_VP, mul((float3x3) unity_ObjectToWorld, norm)); float2 offset = normalize(clipNormal.xy) * _OutlineWidth * distanceFade * clipPosition.w; clipPosition.xy += offset; o.pos = clipPosition; return o; } float4 frag(v2f i) : SV_Target { return _OutlineColor; } ENDHLSL } // 메인 렌더링 패스 (두 번째) Pass { Name "MAIN" Cull Back ZWrite On ZTest LEqual HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" #include "Lighting.cginc" // 메인 셰이더 코드 (이전 셀 셰이딩 코드와 동일) // ... 버텍스/프래그먼트 셰이더 구현 ENDHLSL } } }

✨ 고급 툰 효과

더욱 풍부한 표현을 위한 고급 툰 셰이더 기법들을 구현해보겠습니다.

림 라이팅과 그라디언트 맵

// 고급 툰 셰이더 - 림 라이팅 + 그라디언트 맵 float4 frag(v2f i) : SV_Target { // 기본 설정 float4 albedo = tex2D(_MainTex, i.uv) * _Color; float3 normal = normalize(i.worldNormal); float3 lightDir = normalize(_WorldSpaceLightPos0.xyz); float3 viewDir = normalize(_WorldSpaceCameraPos - i.worldPos); // 기본 라이팅 float NdotL = dot(normal, lightDir); float lightIntensity = smoothstep(_ShadowThreshold - _ShadowSmoothness, _ShadowThreshold + _ShadowSmoothness, NdotL * 0.5 + 0.5); // 그라디언트 맵 적용 float3 gradientColor = tex2D(_GradientMap, float2(lightIntensity, 0.5)).rgb; // 림 라이팅 계산 float NdotV = 1.0 - saturate(dot(normal, viewDir)); float rimIntensity = smoothstep(_RimThreshold - _RimSmoothness, _RimThreshold + _RimSmoothness, NdotV); // 림 색상 계산 (시간 기반 애니메이션) float3 rimColor = _RimColor.rgb * rimIntensity; float rimPulse = sin(_Time.y * _RimPulseSpeed) * 0.5 + 0.5; rimColor *= lerp(1.0, rimPulse, _RimPulseIntensity); // 스페큘러 하이라이트 (툰 스타일) float3 halfVector = normalize(lightDir + viewDir); float NdotH = dot(normal, halfVector); float specular = pow(max(0, NdotH), _Shininess); float specularIntensity = smoothstep(_SpecularThreshold - _SpecularSmoothness, _SpecularThreshold + _SpecularSmoothness, specular); // 하프톤 패턴 (선택적) float2 halftoneUV = i.uv * _HalftoneScale; float halftone = tex2D(_HalftoneTexture, halftoneUV).r; float halftoneEffect = step(lightIntensity, halftone); // 최종 색상 합성 float3 diffuse = albedo.rgb * gradientColor; float3 specularColor = _SpecularColor.rgb * specularIntensity; float3 finalColor = diffuse + specularColor + rimColor; // 하프톤 블렌딩 finalColor = lerp(finalColor, finalColor * halftoneEffect, _HalftoneIntensity); // 앰비언트 추가 finalColor += albedo.rgb * UNITY_LIGHTMODEL_AMBIENT.rgb; return float4(finalColor, albedo.a); }

애니메이션 효과와 인터랙션

// C# 스크립트: 툰 셰이더 애니메이션 컨트롤러 using UnityEngine; public class ToonShaderAnimator : MonoBehaviour { public Material toonMaterial; public AnimationCurve rimPulseCurve = AnimationCurve.EaseInOut(0, 0, 1, 1); public Gradient colorGradient; public float animationSpeed = 1.0f; public bool enableColorAnimation = true; public bool enableRimPulse = true; private float animationTime; private Color originalColor; private Color originalRimColor; void Start() { if (toonMaterial != null) { originalColor = toonMaterial.GetColor("_Color"); originalRimColor = toonMaterial.GetColor("_RimColor"); } } void Update() { if (toonMaterial == null) return; animationTime += Time.deltaTime * animationSpeed; // 색상 애니메이션 if (enableColorAnimation) { float t = Mathf.PingPong(animationTime, 1.0f); Color animatedColor = colorGradient.Evaluate(t); toonMaterial.SetColor("_Color", animatedColor); } // 림 라이팅 펄스 if (enableRimPulse) { float pulseT = Mathf.PingPong(animationTime * 2.0f, 1.0f); float pulseIntensity = rimPulseCurve.Evaluate(pulseT); toonMaterial.SetFloat("_RimPulseIntensity", pulseIntensity); } } // 인터랙션 효과 public void TriggerFlash(Color flashColor, float duration = 0.5f) { StartCoroutine(FlashEffect(flashColor, duration)); } private System.Collections.IEnumerator FlashEffect(Color flashColor, float duration) { Color startColor = toonMaterial.GetColor("_Color"); float elapsed = 0; while (elapsed < duration) { float t = elapsed / duration; float intensity = Mathf.Sin(t * Mathf.PI); Color currentColor = Color.Lerp(startColor, flashColor, intensity); toonMaterial.SetColor("_Color", currentColor); elapsed += Time.deltaTime; yield return null; } toonMaterial.SetColor("_Color", startColor); } void OnDestroy() { // 원래 색상 복원 if (toonMaterial != null) { toonMaterial.SetColor("_Color", originalColor); toonMaterial.SetColor("_RimColor", originalRimColor); } } }

🎯 툰 셰이더 최적화 및 활용

다양한 플랫폼에서 효율적으로 동작하는 툰 셰이더 구현 방법을 알아보겠습니다.

💡 툰 셰이더 최적화 팁:
  • LOD 시스템: 거리별로 아웃라인 두께와 효과 강도 조절
  • 텍스처 최적화: 램프 텍스처와 그라디언트 맵 압축
  • 배칭 친화적: 동일한 머티리얼 사용으로 드로우콜 최소화
  • 모바일 최적화: 복잡한 계산을 버텍스 셰이더로 이동
  • 아웃라인 최적화: 중요한 오브젝트에만 아웃라인 적용
⚠️ 주의사항:
  • 아웃라인 아티팩트: 복잡한 지오메트리에서 아웃라인 깨짐 현상
  • 라이팅 일관성: 씬의 모든 오브젝트에 일관된 툰 스타일 적용
  • 애니메이션 고려: 스키닝된 메시에서 아웃라인 두께 변화
  • 플랫폼별 조정: 모바일과 PC에서 다른 품질 설정
