동적이고 화려한 시각 효과의 핵심

✨ 파티클의 마법

불, 연기, 마법, 폭발 등 모든 동적 효과를 만드는 파티클 셰이더의 세계

🎆 파티클 셰이더의 핵심 요소

파티클 셰이더는 게임에서 가장 화려하고 동적인 시각 효과를 만드는 핵심 기술입니다. 불, 연기, 마법, 폭발, 날씨 효과 등 모든 것을 표현할 수 있습니다.

Basic

기본 파티클

빌보드 렌더링: 항상 카메라를 향하는 스프라이트

텍스처 애니메이션: 프레임 기반 애니메이션

색상 변화: 생명주기별 색상 그라디언트

크기 조절: 시간에 따른 스케일 변화

Animation

애니메이션 효과

UV 스크롤링: 텍스처 이동 효과

회전 애니메이션: 파티클 회전

노이즈 디스토션: 자연스러운 움직임

페이드 인/아웃: 부드러운 등장/소멸

Blending

블렌딩 모드

Additive: 밝은 효과 (불, 마법)

Alpha Blend: 투명 효과 (연기, 구름)

Multiply: 어두운 효과 (그림자)

Screen: 부드러운 발광 효과

Distortion

디스토션 효과

열 왜곡: 뜨거운 공기 효과

물 왜곡: 수면 굴절 효과

마법 왜곡: 공간 비틀림 효과

충격파: 폭발 충격 효과

Volumetric

볼류메트릭 효과

3D 노이즈: 입체적인 구름/연기

레이마칭: 볼륨 렌더링

라이트 스캐터링: 빛 산란 효과

안개 효과: 대기 원근감

GPU

GPU 파티클

컴퓨트 셰이더: GPU 기반 시뮬레이션

대량 파티클: 수만 개 파티클 처리

물리 시뮬레이션: 충돌, 중력, 바람

인스턴싱: 효율적인 렌더링

🔥 기본 파티클 셰이더

가장 기본적인 파티클 셰이더부터 시작해보겠습니다.

기본 빌보드 파티클

Shader "Particle/BasicParticle" { Properties { _MainTex ("Particle Texture", 2D) = "white" {} _TintColor ("Tint Color", Color) = (1, 1, 1, 1) _InvFade ("Soft Particles Factor", Range(0.01, 3.0)) = 1.0 _EmissionStrength ("Emission Strength", Range(0, 10)) = 2.0 } Category { Tags { "Queue"="Transparent" "IgnoreProjector"="True" "RenderType"="Transparent" } Blend SrcAlpha OneMinusSrcAlpha ColorMask RGB Cull Off Lighting Off ZWrite Off SubShader { Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #pragma multi_compile_particles #include "UnityCG.cginc" struct appdata_t { float4 vertex : POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; }; struct v2f { float4 vertex : SV_POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; #ifdef SOFTPARTICLES_ON float4 projPos : TEXCOORD1; #endif }; sampler2D _MainTex; float4 _MainTex_ST; float4 _TintColor; float _InvFade; float _EmissionStrength; #ifdef SOFTPARTICLES_ON sampler2D _CameraDepthTexture; #endif v2f vert(appdata_t v) { v2f o; o.vertex = UnityObjectToClipPos(v.vertex); o.color = v.color; o.texcoord = TRANSFORM_TEX(v.texcoord, _MainTex); #ifdef SOFTPARTICLES_ON o.projPos = ComputeScreenPos(o.vertex); COMPUTE_EYEDEPTH(o.projPos.z); #endif return o; } float4 frag(v2f i) : SV_Target { // 소프트 파티클 계산 #ifdef SOFTPARTICLES_ON float sceneZ = LinearEyeDepth(SAMPLE_DEPTH_TEXTURE_PROJ(_CameraDepthTexture, UNITY_PROJ_COORD(i.projPos))); float partZ = i.projPos.z; float fade = saturate(_InvFade * (sceneZ - partZ)); i.color.a *= fade; #endif // 텍스처 샘플링 float4 tex = tex2D(_MainTex, i.texcoord); // 최종 색상 계산 float4 col = 2.0f * i.color * _TintColor * tex; col.rgb *= _EmissionStrength; return col; } ENDHLSL } } } }

🎨 블렌딩 모드와 효과

다양한 블렌딩 모드를 활용하여 서로 다른 파티클 효과를 만들어보겠습니다.

애니메이션 파티클 셰이더

// 고급 애니메이션 파티클 셰이더 Shader "Particle/AnimatedParticle" { Properties { _MainTex ("Particle Texture", 2D) = "white" {} _TintColor ("Tint Color", Color) = (1, 1, 1, 1) _ScrollSpeed ("Scroll Speed", Vector) = (1, 1, 0, 0) _RotationSpeed ("Rotation Speed", Range(-10, 10)) = 0 _NoiseTexture ("Noise Texture", 2D) = "white" {} _NoiseStrength ("Noise Strength", Range(0, 1)) = 0.1 _DistortionStrength ("Distortion Strength", Range(0, 1)) = 0.1 } Category { Tags { "Queue"="Transparent" "RenderType"="Transparent" } Blend One One // Additive blending Cull Off Lighting Off ZWrite Off SubShader { Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" struct appdata_t { float4 vertex : POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; float4 texcoord1 : TEXCOORD1; // 커스텀 데이터 (생명주기, 회전 등) }; struct v2f { float4 vertex : SV_POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; float2 noiseUV : TEXCOORD1; float lifetime : TEXCOORD2; }; sampler2D _MainTex; sampler2D _NoiseTexture; float4 _MainTex_ST; float4 _TintColor; float4 _ScrollSpeed; float _RotationSpeed; float _NoiseStrength; float _DistortionStrength; // 2D 회전 함수 float2 rotate2D(float2 uv, float angle) { float cosAngle = cos(angle); float sinAngle = sin(angle); float2x2 rotMatrix = float2x2(cosAngle, -sinAngle, sinAngle, cosAngle); return mul(rotMatrix, uv - 0.5) + 0.5; } v2f vert(appdata_t v) { v2f o; o.vertex = UnityObjectToClipPos(v.vertex); o.color = v.color; o.lifetime = v.texcoord1.x; // 생명주기 정보 // UV 스크롤링 float2 scrolledUV = v.texcoord + _ScrollSpeed.xy * _Time.y; // 회전 적용 float rotation = _RotationSpeed * _Time.y + v.texcoord1.y; // 개별 회전 오프셋 o.texcoord = rotate2D(scrolledUV, rotation); // 노이즈 UV o.noiseUV = v.texcoord * 2.0 + _Time.y * 0.5; return o; } float4 frag(v2f i) : SV_Target { // 노이즈 기반 디스토션 float2 noise = tex2D(_NoiseTexture, i.noiseUV).rg * 2.0 - 1.0; float2 distortedUV = i.texcoord + noise * _DistortionStrength; // 메인 텍스처 샘플링 float4 tex = tex2D(_MainTex, distortedUV); // 생명주기 기반 페이드 float lifeFade = smoothstep(0.0, 0.1, i.lifetime) * smoothstep(1.0, 0.8, i.lifetime); // 노이즈 강도 적용 float noiseIntensity = tex2D(_NoiseTexture, i.noiseUV * 0.5).r; float finalAlpha = tex.a * lifeFade * lerp(1.0, noiseIntensity, _NoiseStrength); // 최종 색상 float4 col = tex * i.color * _TintColor; col.a = finalAlpha; return col; } ENDHLSL } } } }

🌪️ 디스토션 효과

열기, 충격파, 마법 등의 공간 왜곡 효과를 구현해보겠습니다.

열 왜곡 셰이더

// 열 왜곡 디스토션 셰이더 Shader "Particle/HeatDistortion" { Properties { _DistortionTex ("Distortion Texture", 2D) = "bump" {} _DistortionStrength ("Distortion Strength", Range(0, 1)) = 0.1 _ScrollSpeed ("Scroll Speed", Vector) = (0, 1, 0, 0) _TintColor ("Tint Color", Color) = (1, 1, 1, 0.5) } Category { Tags { "Queue"="Transparent" "RenderType"="Transparent" } Blend SrcAlpha OneMinusSrcAlpha Cull Off Lighting Off ZWrite Off // 배경을 캡처하기 위한 GrabPass GrabPass { "_BackgroundTexture" } Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" struct appdata_t { float4 vertex : POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; }; struct v2f { float4 vertex : SV_POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; float4 grabPos : TEXCOORD1; }; sampler2D _DistortionTex; sampler2D _BackgroundTexture; float4 _DistortionTex_ST; float _DistortionStrength; float4 _ScrollSpeed; float4 _TintColor; v2f vert(appdata_t v) { v2f o; o.vertex = UnityObjectToClipPos(v.vertex); o.color = v.color; o.texcoord = TRANSFORM_TEX(v.texcoord, _DistortionTex); o.grabPos = ComputeGrabScreenPos(o.vertex); return o; } float4 frag(v2f i) : SV_Target { // 스크롤링 UV float2 scrolledUV = i.texcoord + _ScrollSpeed.xy * _Time.y; // 디스토션 노멀 맵 샘플링 float3 distortionNormal = UnpackNormal(tex2D(_DistortionTex, scrolledUV)); // 다중 레이어 디스토션 float3 distortionNormal2 = UnpackNormal(tex2D(_DistortionTex, scrolledUV * 1.5 + _Time.y * 0.3)); distortionNormal = normalize(distortionNormal + distortionNormal2 * 0.5); // 스크린 좌표 왜곡 float2 distortion = distortionNormal.xy * _DistortionStrength * i.color.a; float4 distortedGrabPos = i.grabPos; distortedGrabPos.xy += distortion * distortedGrabPos.w; // 배경 텍스처 샘플링 float4 backgroundColor = tex2Dproj(_BackgroundTexture, distortedGrabPos); // 열기 효과를 위한 색상 조정 backgroundColor.rgb = lerp(backgroundColor.rgb, backgroundColor.rgb * _TintColor.rgb, _TintColor.a); // 파티클 색상과 블렌딩 float4 finalColor = lerp(backgroundColor, backgroundColor * i.color, i.color.a); return finalColor; } ENDHLSL } } }

☁️ 볼류메트릭 파티클

3D 노이즈를 활용한 입체적인 구름과 연기 효과를 구현해보겠습니다.

3D 노이즈 구름 셰이더

// 볼류메트릭 구름 셰이더 Shader "Particle/VolumetricClouds" { Properties { _CloudColor ("Cloud Color", Color) = (1, 1, 1, 1) _CloudDensity ("Cloud Density", Range(0, 5)) = 1.0 _NoiseScale ("Noise Scale", Range(0.1, 10)) = 1.0 _WindSpeed ("Wind Speed", Vector) = (1, 0, 0.5, 0) _Steps ("Ray Steps", Range(8, 64)) = 32 _StepSize ("Step Size", Range(0.01, 0.5)) = 0.1 } Category { Tags { "Queue"="Transparent" "RenderType"="Transparent" } Blend SrcAlpha OneMinusSrcAlpha Cull Off Lighting Off ZWrite Off Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #include "UnityCG.cginc" struct appdata_t { float4 vertex : POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; }; struct v2f { float4 vertex : SV_POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; float3 worldPos : TEXCOORD1; float3 viewDir : TEXCOORD2; }; float4 _CloudColor; float _CloudDensity; float _NoiseScale; float4 _WindSpeed; int _Steps; float _StepSize; // 3D 노이즈 함수 (Simplex Noise 간소화 버전) float noise3D(float3 p) { float3 i = floor(p); float3 f = frac(p); f = f * f * (3.0 - 2.0 * f); float n = i.x + i.y * 57.0 + 113.0 * i.z; return lerp(lerp(lerp(frac(sin(n) * 43758.5453), frac(sin(n + 1.0) * 43758.5453), f.x), lerp(frac(sin(n + 57.0) * 43758.5453), frac(sin(n + 58.0) * 43758.5453), f.x), f.y), lerp(lerp(frac(sin(n + 113.0) * 43758.5453), frac(sin(n + 114.0) * 43758.5453), f.x), lerp(frac(sin(n + 170.0) * 43758.5453), frac(sin(n + 171.0) * 43758.5453), f.x), f.y), f.z); } // 프랙탈 노이즈 float fbm(float3 p) { float value = 0.0; float amplitude = 0.5; float frequency = 1.0; for (int i = 0; i < 4; i++) { value += amplitude * noise3D(p * frequency); amplitude *= 0.5; frequency *= 2.0; } return value; } v2f vert(appdata_t v) { v2f o; o.vertex = UnityObjectToClipPos(v.vertex); o.color = v.color; o.texcoord = v.texcoord; o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz; o.viewDir = normalize(_WorldSpaceCameraPos - o.worldPos); return o; } float4 frag(v2f i) : SV_Target { // 레이마칭 설정 float3 rayStart = i.worldPos; float3 rayDir = -i.viewDir; float totalDensity = 0.0; float3 lightAccumulation = float3(0, 0, 0); // 레이마칭 루프 for (int step = 0; step < _Steps; step++) { float3 samplePos = rayStart + rayDir * _StepSize * float(step); // 바람 효과 적용 float3 windOffset = _WindSpeed.xyz * _Time.y; float3 noisePos = (samplePos + windOffset) * _NoiseScale; // 노이즈 밀도 계산 float density = fbm(noisePos) * _CloudDensity; density = saturate(density - 0.4) * 2.0; // 임계값 적용 // 거리 기반 페이드 float distanceFade = 1.0 - saturate(float(step) / float(_Steps)); density *= distanceFade; totalDensity += density * _StepSize; // 간단한 라이팅 (태양 방향) float3 lightDir = normalize(float3(1, 1, 1)); float lightDensity = fbm(noisePos + lightDir * 0.5); float lightTransmission = exp(-lightDensity * 2.0); lightAccumulation += density * lightTransmission * _CloudColor.rgb * _StepSize; // 조기 종료 (최적화) if (totalDensity > 0.99) break; } // 최종 색상 계산 float alpha = 1.0 - exp(-totalDensity); float3 finalColor = lightAccumulation / max(totalDensity, 0.001); return float4(finalColor * i.color.rgb, alpha * i.color.a); } ENDHLSL } } }

🎮 GPU 파티클 시스템

컴퓨트 셰이더를 활용한 고성능 GPU 파티클 시스템을 구현해보겠습니다.

컴퓨트 셰이더 파티클 시뮬레이션

// GPU 파티클 컴퓨트 셰이더 #pragma kernel CSMain struct Particle { float3 position; float3 velocity; float4 color; float life; float size; float rotation; float angularVelocity; }; RWStructuredBuffer<Particle> particles; float deltaTime; float3 gravity; float3 wind; float damping; int particleCount; // 노이즈 함수 float random(float2 st) { return frac(sin(dot(st.xy, float2(12.9898, 78.233))) * 43758.5453123); } float3 randomSphere(float seed) { float theta = random(float2(seed, seed + 1.0)) * 2.0 * 3.14159; float phi = acos(2.0 * random(float2(seed + 2.0, seed + 3.0)) - 1.0); float3 dir; dir.x = sin(phi) * cos(theta); dir.y = sin(phi) * sin(theta); dir.z = cos(phi); return dir; } [numthreads(64, 1, 1)] void CSMain(uint3 id : SV_DispatchThreadID) { uint index = id.x; if (index >= uint(particleCount)) return; Particle p = particles[index]; // 생명주기 업데이트 p.life -= deltaTime; // 파티클이 죽었으면 리스폰 if (p.life <= 0.0) { // 새 파티클 초기화 p.position = float3(0, 0, 0); p.velocity = randomSphere(float(index) + _Time.y) * 5.0; p.life = random(float2(float(index), _Time.y)) * 3.0 + 1.0; p.size = random(float2(float(index) + 100, _Time.y)) * 0.5 + 0.5; p.rotation = random(float2(float(index) + 200, _Time.y)) * 6.28; p.angularVelocity = (random(float2(float(index) + 300, _Time.y)) - 0.5) * 10.0; p.color = float4(1, 1, 1, 1); } else { // 물리 시뮬레이션 float3 force = gravity + wind; // 노이즈 기반 터뷸런스 float3 turbulence = float3( sin(p.position.x * 0.1 + _Time.y) * 2.0, cos(p.position.z * 0.1 + _Time.y) * 2.0, sin(p.position.y * 0.1 + _Time.y) * 2.0 ); force += turbulence; // 속도 업데이트 p.velocity += force * deltaTime; p.velocity *= damping; // 위치 업데이트 p.position += p.velocity * deltaTime; // 회전 업데이트 p.rotation += p.angularVelocity * deltaTime; // 생명주기 기반 색상/크기 변화 float lifeRatio = p.life / 3.0; // 최대 생명주기 가정 p.color.a = smoothstep(0.0, 0.1, lifeRatio) * smoothstep(1.0, 0.8, lifeRatio); p.size *= 1.0 + (1.0 - lifeRatio) * 0.5; // 시간이 지날수록 커짐 } particles[index] = p; }

GPU 파티클 렌더링 셰이더

// GPU 파티클 렌더링 셰이더 Shader "Particle/GPUParticleRenderer" { Properties { _MainTex ("Particle Texture", 2D) = "white" {} _TintColor ("Tint Color", Color) = (1, 1, 1, 1) } Category { Tags { "Queue"="Transparent" "RenderType"="Transparent" } Blend One One Cull Off Lighting Off ZWrite Off Pass { HLSLPROGRAM #pragma vertex vert #pragma fragment frag #pragma target 4.5 #include "UnityCG.cginc" struct Particle { float3 position; float3 velocity; float4 color; float life; float size; float rotation; float angularVelocity; }; StructuredBuffer<Particle> particleBuffer; sampler2D _MainTex; float4 _TintColor; struct v2f { float4 vertex : SV_POSITION; float4 color : COLOR; float2 texcoord : TEXCOORD0; }; v2f vert(uint vid : SV_VertexID, uint iid : SV_InstanceID) { v2f o; // 파티클 데이터 가져오기 Particle p = particleBuffer[iid]; // 쿼드 버텍스 생성 (빌보드) float2 quadVertices[4] = { float2(-1, -1), float2(1, -1), float2(-1, 1), float2(1, 1) }; float2 quadUVs[4] = { float2(0, 0), float2(1, 0), float2(0, 1), float2(1, 1) }; float2 vertex = quadVertices[vid]; o.texcoord = quadUVs[vid]; // 회전 적용 float cosRot = cos(p.rotation); float sinRot = sin(p.rotation); float2 rotatedVertex = float2( vertex.x * cosRot - vertex.y * sinRot, vertex.x * sinRot + vertex.y * cosRot ); // 빌보드 변환 float3 worldPos = p.position; float3 cameraRight = normalize(UNITY_MATRIX_V[0].xyz); float3 cameraUp = normalize(UNITY_MATRIX_V[1].xyz); worldPos += (cameraRight * rotatedVertex.x + cameraUp * rotatedVertex.y) * p.size; o.vertex = mul(UNITY_MATRIX_VP, float4(worldPos, 1.0)); o.color = p.color; return o; } float4 frag(v2f i) : SV_Target { float4 tex = tex2D(_MainTex, i.texcoord); return tex * i.color * _TintColor; } ENDHLSL } } }

💡 파티클 최적화 팁:
  • LOD 시스템: 거리별로 파티클 수와 품질 조절
  • 오브젝트 풀링: 파티클 재사용으로 메모리 절약
  • 컬링: 화면 밖 파티클 비활성화
  • 배칭: 같은 머티리얼 파티클 그룹화
  • GPU 인스턴싱: 대량 파티클 효율적 렌더링
⚠️ 주의사항:
